    var AlgebraField = require('./AlgebraField')

function AlgebraElement(field, data) {

  if (! (field instanceof AlgebraField))
    throw new Error()

  this.__defineGetter__('field', function () { return field })

  if (typeof data === 'undefined')
    data = field.one

  this.data = data
}

function addition(element) {
  this.data = this.field.addition(this.data, element.data)
  return this
}
